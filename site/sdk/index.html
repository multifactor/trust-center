const attestation = fs.readFileSync(path.join(__dirname, 'attestation.cbor'))
const attestationResult = await trust.enclaves.nitro.verifyAttestation(attestation)

// verify attestation validity & check that pcr0 matches desired enclave image
if (!attestationResult.valid || attestationResult.attr.pcr0 !== '1595...c963')

// encrypt secret for use within enclave using PGP

What's trust-center?

The Multifactor Trust Center (trust-center) SDK provides tools for interacting with trusted computing devices such as Intel SGX and AWS Nitro enclaves. Validate attestation documents, verify cryptographic proofs, and encrypt secrets for use in enclaves using PGP. Try it out using the online trust center interface at trust.multifactor.com, which is open source, requires no network connectivity, and is hosted on GitHub pages to ensure public auditability.
